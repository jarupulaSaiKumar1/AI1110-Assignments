\section{Question}
\begin{frame}{Question}
    \begin{block}{ICSE 2018 Grade 12 Question viii:} 
        Find the differential equation of the family of concentric circles $x^2+ y^2 = a^2$
    \end{block}
\end{frame}

%Solution
\section{Solution}
\begin{frame}{Solution}
   Given, family of concentric circle is $x^2+ y^2 = a^2$\\
    Here it is given that the circle is concentric.So lets Differentiate w.r.t x
\end{frame}

\begin{frame}{Differentiating w.r.t x we get,}
\begin{align}
2x +2y\frac{dy}{dx} &= 0\\
2y\frac{dy}{dx} &=-2x
\end{align}
this can also be written as,
\begin{align}
\frac{dy}{dx} &=\frac{-2x}{2y}\\
\frac{dy}{dx} &=\frac{-x}{y}
\end{align}
this can also be written as,
\begin{align}
x+y\frac{dy}{dx} &=0
\end{align}
\end{frame}
